using System;
using System.Windows.Forms;

namespace DynamicFormCreation
{
    public partial class DynamicForm : Form
    {
        private Button myButton;

        public DynamicForm(string fimozValue)
        {
            InitializeComponent();

            // Создаем кнопку
            myButton = new Button();
            myButton.Text = "Нажми меня!";
            myButton.Location = new System.Drawing.Point(10, 10);
            myButton.Click += MyButton_Click; // Подписываемся на событие Click
            this.Controls.Add(myButton);
        }

        private void MyButton_Click(object sender, EventArgs e)
        {
            MessageBox.Show("Значение fimoz: ");
        }
    }
}
